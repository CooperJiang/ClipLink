SQL查询

SELECT 
  users.name, 
  orders.order_date, 
  SUM(order_items.price) as total
FROM users
JOIN orders ON users.id = orders.user_id
JOIN order_items ON orders.id = order_items.order_id
WHERE orders.status = 'completed'
GROUP BY users.id, orders.id
ORDER BY total DESC;
